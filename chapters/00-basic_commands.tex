\chapter{Metaadatok es globalis beallitasok}

A dolgozat elejen a \texttt{main.tex}-ben adjuk meg az alapveto adatokat.

\begin{itemize}
	\item \verb|\title{...}| -- a dolgozat cime
	\item \verb|\author{...}| -- a hallgato neve
	\item \verb|\neptun{...}| -- Neptun-kod
	\item \verb|\setdegree{bsc}|, \verb|\setdegree{msc}|, \verb|\setdegree{tdk}|
	\item \verb|\setconfidential{true}| vagy \verb|\setconfidential{false}|
	\item \verb|\program{...}| -- a szak neve
	\item \verb|\specialization{...}| -- szakirany
	\item \verb|\department{...}| -- tanszek
	\item \verb|\supervisor{...}| -- temavezeto neve
	\item \verb|\supervisortitle{...}| -- temavezeto beosztasa
\end{itemize}

\begin{codeblock}{TeX}
	\title{Minta szakdolgozat cime}
	\author{Kiss Peter}
	\neptun{ABC123}
	
	\setdegree{bsc}
	\setconfidential{false}
	
	\program{Programtervezo informatikus}
	\specialization{Szoftverfejleszto}
	\department{Szoftverfejlesztes}
	
	\supervisor{Dr. Nagy Bela}
	\supervisortitle{egyetemi docens}
\end{codeblock}
